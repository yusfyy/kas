<script>
function login() {
    let user = document.getElementById('username').value;
    let pass = document.getElementById('password').value;

    if (user === "admin123" && pass === "admin123") {
        document.getElementById('loginPage').classList.add('hidden');
        document.getElementById('formPage').classList.remove('hidden');
    } else {
        document.getElementById('loginError').style.display = "block";
    }
}

function kirimData() {
    const url = "https://script.google.com/macros/s/AKfycbw0ktcN-XbCyHUDIzSZ4eWCLLKctLWZjlfO8qfDjmdMnLAsGCc80gj7q9gSvEqEeKRf/exec";

    let nama = document.getElementById('nama').value || "-";
    let pemasukan = document.getElementById('pemasukan').value || "-";
    let pengeluaran = document.getElementById('pengeluaran').value || "-";
    let keterangan = document.getElementById('keterangan').value || "-";
    let tanggal = document.getElementById('tanggal').value || "-";

    let data = {
        nama: nama,
        pemasukan: pemasukan,
        pengeluaran: pengeluaran,
        keterangan: keterangan,
        tanggal: tanggal
    };

    fetch(url, {
        method: "POST",
        mode: "no-cors",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(data)
    });

    tambahKeTabel(data);
    kosongkanForm();
}

function tambahKeTabel(data) {
    let tbody = document.querySelector('#dataTable tbody');
    let row = document.createElement('tr');

    row.innerHTML = `
        <td>${data.nama}</td>
        <td>${data.pemasukan}</td>
        <td>${data.pengeluaran}</td>
        <td>${data.keterangan}</td>
        <td>${data.tanggal}</td>
    `;

    tbody.appendChild(row);
}

function kosongkanForm() {
    document.getElementById('nama').value = "";
    document.getElementById('pemasukan').value = "";
    document.getElementById('pengeluaran').value = "";
    document.getElementById('keterangan').value = "";
    document.getElementById('tanggal').value = "";
}
